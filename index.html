import CodeLink_ai

# 初始化CodeLink AI助手
assistant = CodeLink_ai.Assistant()

# 自然语言编程
code = assistant.generate(
    "创建一个快速排序算法",
    language="python"
)

# 智能代码补全
completion = assistant.complete(
    context="def fibonacci(n):",
    cursor_position=15
)

print(f"生成的代码: {code}")
print(f"智能补全: {completion}")
